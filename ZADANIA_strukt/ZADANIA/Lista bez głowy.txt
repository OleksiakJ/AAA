{

//Zliczanie elementow
int zlicz(struct element *Lista){
int licznik=0;

while(Lista!=NULL){
    licznik++;
    Lista=Lista->next;
}

return licznik;
}

//Suma elementow
int suma(struct element *Lista){
int suma=0;

while(Lista!=NULL){
    suma += Lista->i;
    Lista=Lista->next;
}

return suma;
}

//Maksymalny element
int max(struct element *Lista){
int max=Lista->i;

while(Lista!=NULL){
    if(max<Lista){
        max = Lista->i;
    }
     Lista=Lista->next;
}
return max;
}

while(Lista!=NULL){
    if(min>Lista){

     Lista=Lista->next;
